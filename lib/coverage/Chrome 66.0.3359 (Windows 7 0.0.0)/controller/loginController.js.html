define(['app','loginService'],function(app,loginService){
	app.controller('loginController',['$scope','$state','$timeout','loginService','$http','$cookies',function($scope,$state,$timeout,loginService,$http,$cookies){
 
 
		var ctrl = this;
		ctrl.loginStatus = true;
		ctrl.loginObj = {
			'userName':undefined,
			'userPass':undefined
		};
 
		ctrl.waitForLoginToken = function(){
			$timeout(
				function(){
					ctrl.promise
					.then(function(resp){
						ctrl.loginStatus = true;
						console.log(resp.data);
						$cookies.put("userName", ctrl.loginObj.userName);
		                $cookies.put('access_token',resp.data.access_token);
		                loginService.setDefaultHeader();
						$state.go('Home');
					},function(error){
						ctrl.loginStatus = false;
						$state.go('login');
					})
				},1000);
		};
 
		ctrl.chkLogin = function(){
 
			if(ctrl.loginObj['userName'] == undefined){
				window.alert('Please enter user name');
				ctrl.reset();
				return;
			}
 
			else if(ctrl.loginObj['userPass'] == undefined){
				window.alert('please enter password');
				ctrl.reset();
				return;
			}
 
			ctrl.promise = loginService.getToken(ctrl.loginObj);
			ctrl.waitForLoginToken();
                
	}
		ctrl.reset = function(){
			ctrl.loginObj['userName'] = undefined;
			ctrl.loginObj['userPass'] = undefined;
		}
	
	}]);
});
